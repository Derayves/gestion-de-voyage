Conception de la base de données: parking_db (SGBD = Mysql)
(Par TARGOTO CHRISTIAN le 31/03/2021)

create database parking_db character set 'utf8';

use parking_db;

create table tb_place(
num_place int primary key,
type_place varchar(30) not null,
disponible varchar(10) not null)
engine=innodb;

create table tb_occupation(
proprio_id varchar(20) primary key,
proprio_nom varchar(50) not null,
mt_type varchar(30) not null,
mt_matricule varchar(20),
id_place int not null UNIQUE)
engine=innodb;

alter table tb_occupation add constraint fk foreign key(id_place)
references tb_place(num_place);
